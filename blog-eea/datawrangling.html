predictors <- c("gni", "yrs_sch", "life_exp")

# Initialize an empty data frame to store results
results <- data.frame()

# Loop through each predictor variable
for (variable in predictors) {
  # Build a linear regression model
  fit_individual <- lm(gdp ~ get(variable), data = train)
  
  # Make predictions on the test set
  test$prediction_individual <- predict(fit_individual, newdata = select(ungroup(test), !!as.name(variable)))
  
  # Compute and store evaluation metrics
  mse <- mean((test$gdp - test$prediction_individual)^2)
  rmse <- sqrt(mse)
  
  # Create a data frame with results
  result <- data.frame(
    Variable = variable,
    MSE = mse,
    RMSE = rmse
  )
  
  results <- bind_rows(results, result)
  
  # Faceted plot for each variable
  p_individual <- ggplot(test, aes_string(x = variable, y = "gdp", color = "prediction_individual")) +

    geom_smooth(method = "lm", se = FALSE, color = "blue") +
    labs(title = paste("Prediction vs. Actual (", variable, ")", sep = ""),
         x = variable,
         y = "GDP") +
    theme_minimal()
  
  print(p_individual)
}
